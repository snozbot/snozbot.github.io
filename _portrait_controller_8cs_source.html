307  public virtual void RunPortraitCommand(PortraitOptions options, Action onComplete)
308  {
309  waitTimer = 0f;
310 
311  // If no character specified, do nothing
312  if (options.character == null)
313  {
314  onComplete();
315  return;
316  }
317 
318  // If Replace and no replaced character specified, do nothing
319  if (options.display == DisplayType.Replace && options.replacedCharacter == null)
320  {
321  onComplete();
322  return;
323  }
324 
325  // Early out if hiding a character that's already hidden
326  if (options.display == DisplayType.Hide &&
327  !options.character.State.onScreen)
328  {
329  onComplete();
330  return;
331  }
332 
333  options = CleanPortraitOptions(options);
334  options.onComplete = onComplete;
335 
336  switch (options.display)
337  {
338  case (DisplayType.Show):
339  Show(options);
340  break;
341 
342  case (DisplayType.Hide):
343  Hide(options);
344  break;
345 
346  case (DisplayType.Replace):
347  Show(options);
348  Hide(options.replacedCharacter, options.replacedCharacter.State.position.name);
349  break;
350 
351  case (DisplayType.MoveToFront):
352  MoveToFront(options);
353  break;
354  }
355  }

427  public virtual void Show(PortraitOptions options)
428  {
429  options = CleanPortraitOptions(options);
430 
431  if (options.shiftIntoPlace)
432  {
433  options.fromPosition = Instantiate(options.toPosition) as RectTransform;
434  if (options.offset == PositionOffset.OffsetLeft)
435  {
436  options.fromPosition.anchoredPosition =
437  new Vector2(options.fromPosition.anchoredPosition.x - Mathf.Abs(options.shiftOffset.x),
438  options.fromPosition.anchoredPosition.y - Mathf.Abs(options.shiftOffset.y));
439  }
440  else if (options.offset == PositionOffset.OffsetRight)
441  {
442  options.fromPosition.anchoredPosition =
443  new Vector2(options.fromPosition.anchoredPosition.x + Mathf.Abs(options.shiftOffset.x),
444  options.fromPosition.anchoredPosition.y + Mathf.Abs(options.shiftOffset.y));
445  }
446  else
447  {
448  options.fromPosition.anchoredPosition = new Vector2(options.fromPosition.anchoredPosition.x, options.fromPosition.anchoredPosition.y);
449  }
450  }
451 
452  SetupPortrait(options);
453 
454  // LeanTween doesn't handle 0 duration properly
455  float duration = (options.fadeDuration > 0f) ? options.fadeDuration : float.Epsilon;
456 
457  // Fade out a duplicate of the existing portrait image
458  if (options.character.State.portraitImage != null && options.character.State.portraitImage.sprite != null)
459  {
460  GameObject tempGO = GameObject.Instantiate(options.character.State.portraitImage.gameObject);
461  tempGO.transform.SetParent(options.character.State.portraitImage.transform, false);
462  tempGO.transform.localPosition = Vector3.zero;
463  tempGO.transform.localScale = options.character.State.position.localScale;
464 
465  Image tempImage = tempGO.GetComponent<Image>();
466  tempImage.sprite = options.character.State.portraitImage.sprite;
467  tempImage.preserveAspect = true;
468  tempImage.color = options.character.State.portraitImage.color;
469 
470  LeanTween.alpha(tempImage.rectTransform, 0f, duration).setEase(stage.FadeEaseType).setOnComplete(() => {
471  Destroy(tempGO);
472  }).setRecursive(false);
473  }
474 
475  // Fade in the new sprite image
476  if (options.character.State.portraitImage.sprite != options.portrait ||
477  options.character.State.portraitImage.color.a < 1f)
478  {
479  options.character.State.portraitImage.sprite = options.portrait;
480  options.character.State.portraitImage.color = new Color(1f, 1f, 1f, 0f);
481  LeanTween.alpha(options.character.State.portraitImage.rectTransform, 1f, duration).setEase(stage.FadeEaseType).setRecursive(false);
482  }
483 
484  DoMoveTween(options);
485 
486  FinishCommand(options);
487 
488  if (!stage.CharactersOnStage.Contains(options.character))
489  {
490  stage.CharactersOnStage.Add(options.character);
491  }
492 
493  // Update character state after showing
494  options.character.State.onScreen = true;
495  options.character.State.display = DisplayType.Show;
496  options.character.State.portrait = options.portrait;
497  options.character.State.facing = options.facing;
498  options.character.State.position = options.toPosition;
499  }
