74 
75  // Create operator list

87  case CompareOperator.LessThan:

89  break;

92  break;
93  case CompareOperator.LessThanOrEquals:
94  operatorsList.Add(new GUIContent("<="));
95  break;
96  case CompareOperator.GreaterThanOrEquals:
97  operatorsList.Add(new GUIContent(">="));
98  break;
99  default:

101  break;
102  }
103  }
104 
105  // Get previously selected operator
106  int selectedIndex = System.Array.IndexOf(compareOperators, t._CompareOperator);
107  if (selectedIndex < 0)
108  {
109  // Default to first index if the operator is not found in the available operators list
110  // This can occur when changing between variable types
111  selectedIndex = 0;
112  }
113 
114  selectedIndex = EditorGUILayout.Popup(

116  selectedIndex,

120 
121  EditorGUILayout.PropertyField(propByVariableType[variableType]);
