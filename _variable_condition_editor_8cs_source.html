46  public override void DrawCommandGUI()
47  {
48  serializedObject.Update();
49 
50  VariableCondition t = target as VariableCondition;
51 
52  var flowchart = (Flowchart)t.GetFlowchart();
53  if (flowchart == null)
54  {
55  return;
56  }
57 
58  // Select Variable
59  EditorGUILayout.PropertyField(variableProp);
60 
61  if (variableProp.objectReferenceValue == null)
62  {
63  serializedObject.ApplyModifiedProperties();
64  return;
65  }
66 
67  // Get selected variable
68  Variable selectedVariable = variableProp.objectReferenceValue as Variable;
69  System.Type variableType = selectedVariable.GetType();
70 
71  // Get operators for the variable
72  CompareOperator[] compareOperators = VariableCondition.operatorsByVariableType[variableType];
73 
74  // Create operator list
75  List<GUIContent> operatorsList = new List<GUIContent>();
76  foreach (var compareOperator in compareOperators)
77  {
78  switch (compareOperator)
79  {
80  case CompareOperator.Equals:
81  operatorsList.Add(new GUIContent("=="));
82  break;
83  case CompareOperator.NotEquals:
84  operatorsList.Add(new GUIContent("!="));
85  break;
86  case CompareOperator.LessThan:
87  operatorsList.Add(new GUIContent("<"));
88  break;
89  case CompareOperator.GreaterThan:
90  operatorsList.Add(new GUIContent(">"));
91  break;
92  case CompareOperator.LessThanOrEquals:
93  operatorsList.Add(new GUIContent("<="));
94  break;
95  case CompareOperator.GreaterThanOrEquals:
96  operatorsList.Add(new GUIContent(">="));
97  break;
98  default:
99  Debug.LogError("The " + compareOperator.ToString() + " operator has no matching GUIContent.");
100  break;
101  }
102  }
103 
104  // Get previously selected operator
105  int selectedIndex = System.Array.IndexOf(compareOperators, t._CompareOperator);
106  if (selectedIndex < 0)
107  {
108  // Default to first index if the operator is not found in the available operators list
109  // This can occur when changing between variable types
110  selectedIndex = 0;
111  }
112 
113  selectedIndex = EditorGUILayout.Popup(
114  new GUIContent("Compare", "The comparison operator to use when comparing values"),
115  selectedIndex,
116  operatorsList.ToArray());
117 
118  compareOperatorProp.enumValueIndex = (int)compareOperators[selectedIndex];
119 
120  EditorGUILayout.PropertyField(propByVariableType[variableType]);
121 
122  serializedObject.ApplyModifiedProperties();
123  }
