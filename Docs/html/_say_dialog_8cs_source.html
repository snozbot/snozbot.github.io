171  protected virtual void UpdateAlpha()
172  {
173  if (GetWriter().IsWriting)
174  {
175  targetAlpha = 1f;
176  fadeCoolDownTimer = 0.1f;
177  }
178  else if (fadeWhenDone && fadeCoolDownTimer == 0f)
179  {
180  targetAlpha = 0f;
181  }
182  else
183  {
184  // Add a short delay before we start fading in case there's another Say command in the next frame or two.
185  // This avoids a noticeable flicker between consecutive Say commands.
186  fadeCoolDownTimer = Mathf.Max(0f, fadeCoolDownTimer - Time.deltaTime);
187  }
188 
189  CanvasGroup canvasGroup = GetCanvasGroup();
190  if (fadeDuration <= 0f)
191  {
192  canvasGroup.alpha = targetAlpha;
193  }
194  else
195  {
196  float delta = (1f / fadeDuration) * Time.deltaTime;
197  float alpha = Mathf.MoveTowards(canvasGroup.alpha, targetAlpha, delta);
198  canvasGroup.alpha = alpha;
199 
200  if (alpha <= 0f)
201  {
202  // Deactivate dialog object once invisible
203  gameObject.SetActive(false);
204  }
205  }
206  }

373  public virtual IEnumerator DoSay(string text, bool clearPrevious, bool waitForInput, bool fadeWhenDone, bool stopVoiceover, AudioClip voiceOverClip, Action onComplete)
374  {
375  IWriter writer = GetWriter();
376 
377  if (writer.IsWriting || writer.IsWaitingForInput)
378  {
379  writer.Stop();
380  while (writer.IsWriting || writer.IsWaitingForInput)
381  {
382  yield return null;
383  }
384  }
385 
386  gameObject.SetActive(true);
387 
388  this.fadeWhenDone = fadeWhenDone;
389 
390  // Voice over clip takes precedence over a character sound effect if provided
391 
392  AudioClip soundEffectClip = null;
393  if (voiceOverClip != null)
394  {
395  WriterAudio writerAudio = GetWriterAudio();
396  writerAudio.OnVoiceover(voiceOverClip);
397  }
398  else if (speakingCharacter != null)
399  {
400  soundEffectClip = speakingCharacter.SoundEffect;
401  }
402 
403  yield return StartCoroutine(writer.Write(text, clearPrevious, waitForInput, stopVoiceover, soundEffectClip, onComplete));
404  }
