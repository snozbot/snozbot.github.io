36  protected static string[] Split(string stringToSplit)
37  {
38  var results = new List<string>();
39 
40  bool inQuote = false;
41  var currentToken = new StringBuilder();
42  for (int index = 0; index < stringToSplit.Length; ++index)
43  {
44  char currentCharacter = stringToSplit[index];
45  if (currentCharacter == '"')
46  {
47  // When we see a ", we need to decide whether we are
48  // at the start or send of a quoted section...
49  inQuote = !inQuote;
50  }
51  else if (char.IsWhiteSpace(currentCharacter) && !inQuote)
52  {
53  // We've come to the end of a token, so we find the token,
54  // trim it and add it to the collection of results...
55  string result = currentToken.ToString().Trim( new [] { ' ', '\n', '\t', '\"'} );
56  if (result != "") results.Add(result);
57 
58  // We start a new token...
59  currentToken = new StringBuilder();
60  }
61  else
62  {
63  // We've got a 'normal' character, so we add it to
64  // the curent token...
65  currentToken.Append(currentCharacter);
66  }
67  }
68 
69  // We've come to the end of the string, so we add the last token...
70  string lastResult = currentToken.ToString().Trim();
71  if (lastResult != "")
72  {
73  results.Add(lastResult);
74  }
75 
76  return results.ToArray();
77  }

98  protected virtual List<ConversationItem> Parse(string conv)
99  {
100  //find SimpleScript say strings with portrait options
101  //You can test regex matches here: http://regexstorm.net/tester
102  var sayRegex = new Regex(@"((?<sayParams>[\w ""><.']*):)?(?<text>.*)\r*(\n|$)");
103  MatchCollection sayMatches = sayRegex.Matches(conv);
104 
105  var items = new List<ConversationItem>(sayMatches.Count);
106 
107  Character currentCharacter = null;
108  for (int i = 0; i < sayMatches.Count; i++)
109  {
110  string text = sayMatches[i].Groups["text"].Value.Trim();
111  string sayParams = sayMatches[i].Groups["sayParams"].Value;
112 
113  // As text and SayParams are both optional, an empty string will match the regex.
114  // We can ignore any matches where both are empty
115  // or if they're Lua style comments
116  if ((text.Length == 0 && sayParams.Length == 0) || text.StartsWith("--"))
117  {
118  continue;
119  }
120 
121  string[] separateParams = null;
122 
123  if (!string.IsNullOrEmpty(sayParams))
124  {
125  separateParams = Split(sayParams);
126  }
127 
128  var item = CreateConversationItem(separateParams, text, currentCharacter);
129 
130  // Previous speaking character is the default for next conversation item
131  currentCharacter = item.Character;
132 
133  items.Add(item);
134  }
135 
136  return items;
137  }

147  protected virtual ConversationItem CreateConversationItem(string[] sayParams, string text, Character currentCharacter)
148  {
149  var item = new ConversationItem();
150 
151  // Populate the story text to be written
152  item.Text = text;
153 
154  if (sayParams == null || sayParams.Length == 0)
155  {
156  // Text only, no params - early out.
157  return item;
158  }
159 
160  // try to find the character param first, since we need to get its portrait
161  int characterIndex = -1;
162  if (characters == null)
163  {
164  PopulateCharacterCache();
165  }
166 
167  for (int i = 0; item.Character == null && i < sayParams.Length; i++)
168  {
169  for (int j = 0; j < characters.Length; j++)
170  {
171  if (characters[j].NameStartsWith(sayParams[i]))
172  {
173  characterIndex = i;
174  item.Character = characters[j];
175  break;
176  }
177  }
178  }
179 
180  // Assume last used character if none is specified now
181  if (item.Character == null)
182  {
183  item.Character = currentCharacter;
184  }
185 
186  // Check if there's a Hide parameter
187  int hideIndex = -1;
188  if (item.Character != null)
189  {
190  for (int i = 0; i < sayParams.Length; i++)
191  {
192  if (i != characterIndex &&
193  string.Compare(sayParams[i], "hide", true) == 0 )
194  {
195  hideIndex = i;
196  item.Hide = true;
197  break;
198  }
199  }
200  }
201 
202  int flipIndex = -1;
203  if (item.Character != null)
204  {
205  for (int i = 0; i < sayParams.Length; i++)
206  {
207  if (i != characterIndex &&
208  i != hideIndex &&
209  (string.Compare(sayParams[i], ">>>", true) == 0
210  || string.Compare(sayParams[i], "<<<", true) == 0))
211  {
212  if (string.Compare(sayParams[i], ">>>", true) == 0) item.FacingDirection = FacingDirection.Right;
213  if (string.Compare(sayParams[i], "<<<", true) == 0) item.FacingDirection = FacingDirection.Left;
214  flipIndex = i;
215  item.Flip = true;
216  break;
217  }
218  }
219  }
220 
221  // Next see if we can find a portrait for this character
222  int portraitIndex = -1;
223  if (item.Character != null)
224  {
225  for (int i = 0; i < sayParams.Length; i++)
226  {
227  if (item.Portrait == null &&
228  item.Character != null &&
229  i != characterIndex &&
230  i != hideIndex &&
231  i != flipIndex)
232  {
233  Sprite s = item.Character.GetPortrait(sayParams[i]);
234  if (s != null)
235  {
236  portraitIndex = i;
237  item.Portrait = s;
238  break;
239  }
240  }
241  }
242  }
243 
244  // Next check if there's a position parameter
245  Stage stage = Stage.GetActiveStage();
246  if (stage != null)
247  {
248  for (int i = 0; i < sayParams.Length; i++)
249  {
250  if (i != characterIndex &&
251  i != portraitIndex &&
252  i != hideIndex)
253  {
254  RectTransform r = stage.GetPosition(sayParams[i]);
255  if (r != null)
256  {
257  item.Position = r;
258  break;
259  }
260  }
261  }
262  }
263 
264  return item;
265  }

281  public virtual IEnumerator DoConversation(string conv)
282  {
283  if (string.IsNullOrEmpty(conv))
284  {
285  yield break;
286  }
287 
288  var conversationItems = Parse(conv);
289 
290  if (conversationItems.Count == 0)
291  {
292  yield break;
293  }
294 
295  // Track the current and previous parameter values
296  Character currentCharacter = null;
297  Sprite currentPortrait = null;
298  RectTransform currentPosition = null;
299  Character previousCharacter = null;
300 
301  // Play the conversation
302  for (int i = 0; i < conversationItems.Count; ++i)
303  {
304  ConversationItem item = conversationItems[i];
305 
306  if (item.Character != null)
307  {
308  currentCharacter = item.Character;
309  }
310 
311  currentPortrait = item.Portrait;
312  currentPosition = item.Position;
313 
314  var sayDialog = GetSayDialog(currentCharacter);
315 
316  if (sayDialog == null)
317  {
318  // Should never happen
319  yield break;
320  }
321 
322  sayDialog.SetActive(true);
323 
324  if (currentCharacter != null &&
325  currentCharacter != previousCharacter)
326  {
327  sayDialog.SetCharacter(currentCharacter);
328  }
329 
330  var stage = Stage.GetActiveStage();
331 
332  if (stage != null && currentCharacter != null &&
333  (currentPortrait != currentCharacter.State.portrait ||
334  currentPosition != currentCharacter.State.position))
335  {
336  var portraitOptions = new PortraitOptions(true);
337  portraitOptions.display = item.Hide ? DisplayType.Hide : DisplayType.Show;
338  portraitOptions.character = currentCharacter;
339  portraitOptions.fromPosition = currentCharacter.State.position;
340  portraitOptions.toPosition = currentPosition;
341  portraitOptions.portrait = currentPortrait;
342 
343  //Flip option - Flip the opposite direction the character is currently facing
344  if (item.Flip) portraitOptions.facing = item.FacingDirection;
345 
346  // Do a move tween if the character is already on screen and not yet at the specified position
347  if (currentCharacter.State.onScreen &&
348  currentPosition != currentCharacter.State.position)
349  {
350  portraitOptions.move = true;
351  }
352 
353  if (item.Hide)
354  {
355  stage.Hide(portraitOptions);
356  }
357  else
358  {
359  stage.Show(portraitOptions);
360  }
361  }
362 
363  if (stage == null &&
364  currentPortrait != null)
365  {
366  sayDialog.SetCharacterImage(currentPortrait);
367  }
368 
369  previousCharacter = currentCharacter;
370 
371  if (!string.IsNullOrEmpty(item.Text)) {
372  exitSayWait = false;
373  sayDialog.Say(item.Text, true, true, true, false, null, () => {
374  exitSayWait = true;
375  });
376 
377  while (!exitSayWait)
378  {
379  yield return null;
380  }
381  exitSayWait = false;
382  }
383  }
384  }
385 
386  #endregion
387  }
