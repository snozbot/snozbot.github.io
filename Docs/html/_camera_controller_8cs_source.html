91  {
92  float startAlpha = fadeAlpha;
93  float timer = 0;
94 
95  // If already at the target alpha then complete immediately
96  if (Mathf.Approximately(startAlpha, targetAlpha))
97  {
98  yield return null;
99  }
100  else
101  {
102  while (timer < fadeDuration)
103  {
104  float t = timer / fadeDuration;
105  timer += Time.deltaTime;

109  fadeAlpha = Mathf.Lerp(startAlpha, targetAlpha, t);

111  }
112  }
113

115 
116  if (fadeAction != null)
117  {

122  protected virtual IEnumerator PanInternal(Camera camera, Vector3 targetPos, Quaternion targetRot, float targetSize, float duration, Action arriveAction)
123  {
124  if (camera == null)
125  {
126  Debug.LogWarning("Camera is null");
127  yield break;
128  }
129 
130  float timer = 0;
131  float startSize = camera.orthographicSize;
132  float endSize = targetSize;
133  Vector3 startPos = camera.transform.position;
134  Vector3 endPos = targetPos;
135  Quaternion startRot = camera.transform.rotation;
136  Quaternion endRot = targetRot;
137 
138  bool arrived = false;
139  while (!arrived)
140  {
141  timer += Time.deltaTime;
142  if (timer > duration)
143  {
144  arrived = true;
145  timer = duration;
146  }
147 
148  // Apply smoothed lerp to camera position and orthographic size
149  float t = 1f;
150  if (duration > 0f)
151  {
152  t = timer / duration;
153  }
154 
155  if (camera != null)
156  {
157  camera.orthographicSize = Mathf.Lerp(startSize, endSize, Mathf.SmoothStep(0f, 1f, t));
158  camera.transform.position = Vector3.Lerp(startPos, endPos, Mathf.SmoothStep(0f, 1f, t));
159  camera.transform.rotation = Quaternion.Lerp(startRot, endRot, Mathf.SmoothStep(0f, 1f, t));
160  }
161 
162  SetCameraZ(camera);
163 
164  if (arrived &&
165  arriveAction != null)
166  {
167  arriveAction();
168  }
169 
170  yield return null;
171  }
172  }

269  protected virtual Vector3 CalcCameraPosition(Vector3 pos, View viewA, View viewB)
270  {
271  Vector3 safePos = pos;

273  // Clamp camera position to region defined by the two views
274  safePos.x = Mathf.Max(safePos.x, Mathf.Min(viewA.transform.position.x, viewB.transform.position.x));
275  safePos.x = Mathf.Min(safePos.x, Mathf.Max(viewA.transform.position.x, viewB.transform.position.x));
276  safePos.y = Mathf.Max(safePos.y, Mathf.Min(viewA.transform.position.y, viewB.transform.position.y));

282  // Smoothly interpolate camera orthographic size based on relative position to two views
283  protected virtual float CalcCameraSize(Vector3 pos, View viewA, View viewB)
284  {
285  // Get ray and point in same space
286  Vector3 toViewB = viewB.transform.position - viewA.transform.position;
287  Vector3 localPos = pos - viewA.transform.position;
288 
289  // Normalize
290  float distance = toViewB.magnitude;
291  toViewB /= distance;
292  localPos /= distance;
293 
294  // Project point onto ray
295  float t = Vector3.Dot(toViewB, localPos);
296  t = Mathf.Clamp01(t); // Not really necessary but no harm
297 
298  float cameraSize = Mathf.Lerp(viewA.ViewSize, viewB.ViewSize, t);
299 
300  return cameraSize;
301  }
