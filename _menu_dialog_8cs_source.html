219  public virtual bool AddOption(string text, bool interactable, Block targetBlock)
220  {
221  bool addedOption = false;
222  for (int i = 0; i < cachedButtons.Length; i++)
223  {
224  var button = cachedButtons[i];
225  if (!button.gameObject.activeSelf)
226  {
227  button.gameObject.SetActive(true);
228  button.interactable = interactable;
229  if (interactable && autoSelectFirstButton && !cachedButtons.Select(x => x.gameObject).Contains(EventSystem.current.currentSelectedGameObject))
230  {
231  EventSystem.current.SetSelectedGameObject(button.gameObject);
232  }
233  Text textComponent = button.GetComponentInChildren<Text>();
234  if (textComponent != null)
235  {
236  textComponent.text = text;
237  }
238  var block = targetBlock;
239  button.onClick.AddListener(delegate
240  {
241  EventSystem.current.SetSelectedGameObject(null);
242  StopAllCoroutines();
243  // Stop timeout
244  Clear();
245  HideSayDialog();
246  if (block != null)
247  {
248  var flowchart = block.GetFlowchart();
249  #if UNITY_EDITOR
250  // Select the new target block in the Flowchart window
251  flowchart.SelectedBlock = block;
252  #endif
253  gameObject.SetActive(false);
254  // Use a coroutine to call the block on the next frame
255  // Have to use the Flowchart gameobject as the MenuDialog is now inactive
256  flowchart.StartCoroutine(CallBlock(block));
257  }
258  });
259  addedOption = true;
260  break;
261  }
262  }
263 
264  return addedOption;
265  }
