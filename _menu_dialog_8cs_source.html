200  public virtual bool AddOption(string text, bool interactable, Block targetBlock)
201  {
202  bool addedOption = false;
203  for (int i = 0; i < cachedButtons.Length; i++)
204  {
205  var button = cachedButtons[i];
206  if (!button.gameObject.activeSelf)
207  {
208  button.gameObject.SetActive(true);
209  button.interactable = interactable;
210  if (interactable && autoSelectFirstButton && !cachedButtons.Select(x => x.gameObject).Contains(EventSystem.current.currentSelectedGameObject))
211  {
212  EventSystem.current.SetSelectedGameObject(button.gameObject);
213  }
214  Text textComponent = button.GetComponentInChildren<Text>();
215  if (textComponent != null)
216  {
217  textComponent.text = text;
218  }
219  var block = targetBlock;
220  button.onClick.AddListener(delegate
221  {
222  EventSystem.current.SetSelectedGameObject(null);
223  StopAllCoroutines();
224  // Stop timeout
225  Clear();
226  HideSayDialog();
227  if (block != null)
228  {
229  var flowchart = block.GetFlowchart();
230  #if UNITY_EDITOR
231  // Select the new target block in the Flowchart window
232  flowchart.SelectedBlock = block;
233  #endif
234  gameObject.SetActive(false);
235  // Use a coroutine to call the block on the next frame
236  // Have to use the Flowchart gameobject as the MenuDialog is now inactive
237  flowchart.StartCoroutine(CallBlock(block));
238  }
239  });
240  addedOption = true;
241  break;
242  }
243  }
244 
245  return addedOption;
246  }
