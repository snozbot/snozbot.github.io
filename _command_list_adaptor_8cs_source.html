248  // There's a weird incompatibility between the Reorderable list control used for the command list and
249  // the UnityEvent list control used in some commands. In play mode, if you click on the reordering grabber
250  // for a command in the list it causes the UnityEvent list to spew null exception errors.
251  // The workaround for now is to hide the reordering grabber from mouse clicks by extending the command
252  // selection rectangle to cover it. We are planning to totally replace the command list display system.
253  Rect clickRect = position;
254  clickRect.x -= 20;
255  clickRect.width += 20;
256 
257  // Select command via left click
258  if (Event.current.type == EventType.MouseDown &&
259  Event.current.button == 0 &&
260  clickRect.Contains(Event.current.mousePosition))
261  {
262  if (flowchart.SelectedCommands.Contains(command) && Event.current.button == 0)
263  {
264  // Left click on already selected command
265  // Command key and shift key is not pressed
266  if (!EditorGUI.actionKey && !Event.current.shift)
267  {
268  BlockEditor.actionList.Add ( delegate {

273

277  BlockEditor.actionList.Add ( delegate {

304  {

308  {

320  {

322  foreach (Command selectedCommand in flowchart.SelectedCommands)

342  lastSelectedIndex = currentIndex;
343  }
344  else
345  {

348  firstSelectedIndex = currentIndex;
349  }
350  if (currentIndex > lastSelectedIndex)

352  lastSelectedIndex = currentIndex;
353  }
354  }
355

357  {
358  var selectedCommand = flowchart.SelectedBlock.CommandList[i];
359  BlockEditor.actionList.Add ( delegate {
360  flowchart.AddSelectedCommand(selectedCommand);
361  });
362  }
363  }

365  Event.current.Use();
366  }
367  GUIUtility.keyboardControl = 0; // Fix for textarea not refeshing (change focus)
