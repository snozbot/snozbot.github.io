19  protected static bool clip_test(float p, float q, ref float u1, ref float u2)
20  {
21  float r;
22  bool retval = true;
23 
24  if (p < 0.0)
25  {
26  r = q / p;
27  if (r > u2)
28  retval = false;
29  else if (r > u1)
30  u1 = r;
31  }
32  else if (p > 0.0)
33  {
34  r = q / p;

36  retval = false;
37  else if (r < u2)
38  u2 = r;
39  }
40  else if (q < 0.0)
41  retval = false;
42 
43  return retval;

46  public static bool segment_rect_intersection(Rect bounds, ref Vector2 p1, ref Vector2 p2)
47  {
48  float u1 = 0.0f, u2 = 1.0f, dx = p2.x - p1.x, dy;
49 
50  if (clip_test(-dx, p1.x - bounds.xMin, ref u1, ref u2))
51  {
52  if (clip_test(dx, bounds.xMax - p1.x, ref u1, ref u2))
53  {
54  dy = p2.y - p1.y;
55  if (clip_test(-dy, p1.y - bounds.yMin, ref u1, ref u2))
56  {
57  if (clip_test(dy, bounds.yMax - p1.y, ref u1, ref u2))
58  {
59  if (u2 < 1.0)
60  {
61  p2.x = p1.x + u2 * dx;
62  p2.y = p1.y + u2 * dy;
63  }
64 
65  if (u1 > 0.0)
66  {
67  p1.x += u1 * dx;
68  p1.y += u1 * dy;
69  }
70  return true;
71  }
72  }
73  }
74  }

285  private static Vector2 CubeBezier(Vector2 s, Vector2 st, Vector2 e, Vector2 et, float t)
286  {
287  float rt = 1 - t;
288  float rtt = rt * t;
289  return rt * rt * rt * s + 3 * rt * rtt * st + 3 * rtt * t * et + t * t * t * e;
