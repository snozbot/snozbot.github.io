89  {
90  float startAlpha = fadeAlpha;

92

120  protected virtual IEnumerator PanInternal(Camera camera, Vector3 targetPos, Quaternion targetRot, float targetSize, float duration, Action arriveAction)
121  {
122  if (camera == null)
123  {
124  Debug.LogWarning("Camera is null");
125  yield break;
126  }
127 
128  float timer = 0;
129  float startSize = camera.orthographicSize;
130  float endSize = targetSize;
131  Vector3 startPos = camera.transform.position;
132  Vector3 endPos = targetPos;
133  Quaternion startRot = camera.transform.rotation;
134  Quaternion endRot = targetRot;
135 
136  bool arrived = false;
137  while (!arrived)
138  {
139  timer += Time.deltaTime;
140  if (timer > duration)
141  {
142  arrived = true;
143  timer = duration;
144  }
145 
146  // Apply smoothed lerp to camera position and orthographic size
147  float t = 1f;
148  if (duration > 0f)
149  {
150  t = timer / duration;
151  }
152 
153  if (camera != null)
154  {
155  camera.orthographicSize = Mathf.Lerp(startSize, endSize, Mathf.SmoothStep(0f, 1f, t));
156  camera.transform.position = Vector3.Lerp(startPos, endPos, Mathf.SmoothStep(0f, 1f, t));
157  camera.transform.rotation = Quaternion.Lerp(startRot, endRot, Mathf.SmoothStep(0f, 1f, t));
158 
159  SetCameraZ(camera);
160  }
161 
162  if (arrived &&
163  arriveAction != null)
164  {
165  arriveAction();
166  }
167 
168  yield return null;
169  }
170  }

280  // Smoothly interpolate camera orthographic size based on relative position to two views
281  protected virtual float CalcCameraSize(Vector3 pos, View viewA, View viewB)
282  {
283  // Get ray and point in same space
284  Vector3 toViewB = viewB.transform.position - viewA.transform.position;
285  Vector3 localPos = pos - viewA.transform.position;
286 
287  // Normalize
288  float distance = toViewB.magnitude;
289  toViewB /= distance;
290  localPos /= distance;
291 
292  // Project point onto ray
293  float t = Vector3.Dot(toViewB, localPos);
294  t = Mathf.Clamp01(t); // Not really necessary but no harm
295 
296  float cameraSize = Mathf.Lerp(viewA.ViewSize, viewB.ViewSize, t);
297 
298  return cameraSize;
299  }
