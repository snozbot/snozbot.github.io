140  protected virtual void UpdateAlpha()
141  {
142  if (GetWriter().IsWriting)
143  {
144  targetAlpha = 1f;
145  fadeCoolDownTimer = 0.1f;
146  }
147  else if (fadeWhenDone && Mathf.Approximately(fadeCoolDownTimer, 0f))
148  {
149  targetAlpha = 0f;
150  }
151  else
152  {
153  // Add a short delay before we start fading in case there's another Say command in the next frame or two.
154  // This avoids a noticeable flicker between consecutive Say commands.
155  fadeCoolDownTimer = Mathf.Max(0f, fadeCoolDownTimer - Time.deltaTime);
156  }
157 
158  CanvasGroup canvasGroup = GetCanvasGroup();
159  if (fadeDuration <= 0f)
160  {
161  canvasGroup.alpha = targetAlpha;
162  }
163  else
164  {
165  float delta = (1f / fadeDuration) * Time.deltaTime;
166  float alpha = Mathf.MoveTowards(canvasGroup.alpha, targetAlpha, delta);
167  canvasGroup.alpha = alpha;
168 
169  if (alpha <= 0f)
170  {
171  // Deactivate dialog object once invisible
172  gameObject.SetActive(false);
173  }
174  }
175  }

419  public virtual IEnumerator DoSay(string text, bool clearPrevious, bool waitForInput, bool fadeWhenDone, bool stopVoiceover, AudioClip voiceOverClip, Action onComplete)
420  {
421  var writer = GetWriter();
422 
423  if (writer.IsWriting || writer.IsWaitingForInput)
424  {
425  writer.Stop();
426  while (writer.IsWriting || writer.IsWaitingForInput)
427  {
428  yield return null;
429  }
430  }
431 
432  gameObject.SetActive(true);
433 
434  this.fadeWhenDone = fadeWhenDone;
435 
436  // Voice over clip takes precedence over a character sound effect if provided
437 
438  AudioClip soundEffectClip = null;
439  if (voiceOverClip != null)
440  {
441  WriterAudio writerAudio = GetWriterAudio();
442  writerAudio.OnVoiceover(voiceOverClip);
443  }
444  else if (speakingCharacter != null)
445  {
446  soundEffectClip = speakingCharacter.SoundEffect;
447  }
448 
449  yield return StartCoroutine(writer.Write(text, clearPrevious, waitForInput, stopVoiceover, soundEffectClip, onComplete));
450  }
