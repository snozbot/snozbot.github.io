1189  {

1191  new Vector2(rectA.xMin, rectA.center.y),
1192  new Vector2(rectA.xMin + rectA.width / 2, rectA.yMin),
1193  new Vector2(rectA.xMin + rectA.width / 2, rectA.yMax),
1194  new Vector2(rectA.xMax, rectA.center.y)

1198  new Vector2(rectB.xMin, rectB.center.y),
1199  new Vector2(rectB.xMin + rectB.width / 2, rectB.yMin),
1200  new Vector2(rectB.xMin + rectB.width / 2, rectB.yMax),
1201  new Vector2(rectB.xMax, rectB.center.y)
1202  };
1203 
1204  Vector2 pointA = Vector2.zero;
1205  Vector2 pointB = Vector2.zero;
1206  float minDist = float.MaxValue;
1207 
1208  foreach (var a in pointsA)
1209  {
1210  foreach (var b in pointsB)
1211  {
1212  float d = Vector2.Distance(a, b);
1213  if (d < minDist)
1214  {
1215  pointA = a;
1216  pointB = b;
1217  minDist = d;
1218  }

1223  if (highlight)
1224  {
1225  color = Color.green;
1226  }
1227 
1228  Handles.color = color;
1229 
1230  // Place control based on distance between points
1231  // Weight the min component more so things don't get overly curvy
1232  var diff = pointA - pointB;
1233  diff.x = Mathf.Abs(diff.x);
1234  diff.y = Mathf.Abs(diff.y);
1235  var min = Mathf.Min(diff.x, diff.y);
1236  var max = Mathf.Max(diff.x, diff.y);
1237  var mod = min * 0.75f + max * 0.25f;
1238

1240  var directionA = (rectA.center - pointA).normalized;
1241  var directionB = (rectB.center - pointB).normalized;

1243  var controlB = pointB - directionB * mod * 0.67f;

1255  var connectionRectA = new Rect(connectionPointA.x - 4f, connectionPointA.y - 4f, 8f, 8f);
1256  var connectionPointB = pointB + directionB * 4f;
1257  var connectionRectB = new Rect(connectionPointB.x - 4f, connectionPointB.y - 4f, 8f, 8f);

1259  GUI.DrawTexture(connectionRectA, connectionPointTexture, ScaleMode.ScaleToFit);
