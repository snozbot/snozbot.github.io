411  public virtual void CalcFlowchartCenter(Flowchart flowchart, Block[] blocks)
412  {
413  if (flowchart == null ||
414  blocks.Count() == 0)
415  {
416  return;
417  }
418 
419  Vector2 min = blocks[0]._NodeRect.min;
420  Vector2 max = blocks[0]._NodeRect.max;
421 
422  foreach (var block in blocks)
423  {
424  min.x = Mathf.Min(min.x, block._NodeRect.center.x);
425  min.y = Mathf.Min(min.y, block._NodeRect.center.y);
426  max.x = Mathf.Max(max.x, block._NodeRect.center.x);
427  max.y = Mathf.Max(max.y, block._NodeRect.center.y);
428  }
429 
430  Vector2 center = (min + max) * -0.5f;
431 
432  center.x += position.width * 0.5f;
433  center.y += position.height * 0.5f;
434 
435  flowchart.CenterPosition = center;
436  }
