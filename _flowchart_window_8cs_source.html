412  public virtual void CalcFlowchartCenter(Flowchart flowchart, Block[] blocks)
413  {
414  if (flowchart == null ||
415  blocks.Count() == 0)
416  {
417  return;
418  }
419 
420  Vector2 min = blocks[0]._NodeRect.min;
421  Vector2 max = blocks[0]._NodeRect.max;
422 
423  foreach (var block in blocks)
424  {
425  min.x = Mathf.Min(min.x, block._NodeRect.center.x);
426  min.y = Mathf.Min(min.y, block._NodeRect.center.y);
427  max.x = Mathf.Max(max.x, block._NodeRect.center.x);
428  max.y = Mathf.Max(max.y, block._NodeRect.center.y);
429  }
430 
431  Vector2 center = (min + max) * -0.5f;
432 
433  center.x += position.width * 0.5f;
434  center.y += position.height * 0.5f;
435 
436  flowchart.CenterPosition = center;
437  }
