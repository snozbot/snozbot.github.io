479  if (e.button == 0 && searchRect.Contains(e.mousePosition))
480  {
481  blockPopupSelection = 0;
482  popupScroll = Vector2.zero;
483  }
484 
485  rightClickDown = -Vector2.one;
486  break;
487 
488  case EventType.KeyDown:
489  if (GUI.GetNameOfFocusedControl() == searchFieldName)
490  {
491  var centerBlock = false;
492  var selectBlock = false;
493  var closePopup = false;
494  var useEvent = false;
495 
496  switch (e.keyCode)
497  {
498  case KeyCode.DownArrow:
499  ++blockPopupSelection;
500  centerBlock = true;
501  useEvent = true;
502  break;
503 
504  case KeyCode.UpArrow:
505  --blockPopupSelection;
506  centerBlock = true;
507  useEvent = true;
508  break;
509 
510  case KeyCode.Return:
511  centerBlock = true;
512  selectBlock = true;
513  closePopup = true;
514  useEvent = true;
515  break;
516 
517  case KeyCode.Escape:
518  closePopup = true;
519  useEvent = true;
520  break;
521  }
522 
523  blockPopupSelection = Mathf.Clamp(blockPopupSelection, 0, filteredBlocks.Length - 1);
524 
525  if (centerBlock && filteredBlocks.Length > 0)
526  {
527  var block = filteredBlocks[blockPopupSelection];
528  CenterBlock(block);
529 
530  if (selectBlock)
531  {
532  SelectBlock(block);
533  }
534  }
535 
536  if (closePopup)
537  {
538  CloseBlockPopup();
539  }
540 
541  if (useEvent)
542  {
543  e.Use();
544  }
545  }
546  else if (e.keyCode == KeyCode.Escape && flowchart.SelectedBlocks.Count > 0)
547  {
548  DeselectAll();
549  e.Use();
550  }
551  else if (e.control && !wasControl)
552  {
553  StartControlSelection();
554  Repaint();

556  }
557  break;
558  case EventType.KeyUp:
559  if (!e.control && wasControl)

564  }
