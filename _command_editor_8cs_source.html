128  // Users should not be able to change the MonoScript for the command using the usual Script field.
129  // Doing so could cause block.commandList to contain null entries.
130  // To avoid this we manually display all properties, except for m_Script.
131  serializedObject.Update();
132  SerializedProperty iterator = serializedObject.GetIterator();
133  bool enterChildren = true;

164  public static void ObjectField<T>(SerializedProperty property, GUIContent label, GUIContent nullLabel, List<T> objectList) where T : Object
165  {
166  if (property == null)
167  {
168  return;
169  }
170 
171  List<GUIContent> objectNames = new List<GUIContent>();
172 
173  T selectedObject = property.objectReferenceValue as T;
174 
175  int selectedIndex = -1; // Invalid index
176 
177  // First option in list is <None>
178  objectNames.Add(nullLabel);
179  if (selectedObject == null)
180  {
181  selectedIndex = 0;
182  }
183 
184  for (int i = 0; i < objectList.Count; ++i)
185  {
186  if (objectList[i] == null) continue;
187  objectNames.Add(new GUIContent(objectList[i].name));
188 
189  if (selectedObject == objectList[i])
190  {
191  selectedIndex = i + 1;
192  }
193  }
194 
195  T result;
196 
197  selectedIndex = EditorGUILayout.Popup(label, selectedIndex, objectNames.ToArray());
198 
199  if (selectedIndex == -1)
200  {
201  // Currently selected object is not in list, but nothing else was selected so no change.
202  return;
203  }
204  else if (selectedIndex == 0)
205  {
206  result = null; // Null option
207  }
208  else
209  {
210  result = objectList[selectedIndex - 1];
211  }
212 
213  property.objectReferenceValue = result;
214  }
