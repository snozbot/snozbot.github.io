123  // Users should not be able to change the MonoScript for the command using the usual Script field.
124  // Doing so could cause block.commandList to contain null entries.

159  public static void ObjectField<T>(SerializedProperty property, GUIContent label, GUIContent nullLabel, List<T> objectList) where T : Object
160  {
161  if (property == null)

166  List<GUIContent> objectNames = new List<GUIContent>();
167 
168  T selectedObject = property.objectReferenceValue as T;
169 
170  int selectedIndex = -1; // Invalid index
171 
172  // First option in list is <None>
173  objectNames.Add(nullLabel);

176  selectedIndex = 0;
177  }
178 
179  for (int i = 0; i < objectList.Count; ++i)

182  objectNames.Add(new GUIContent(objectList[i].name));
183 
184  if (selectedObject == objectList[i])
185  {
186  selectedIndex = i + 1;
187  }
188  }
189 
190  T result;
191 
192  selectedIndex = EditorGUILayout.Popup(label, selectedIndex, objectNames.ToArray());
193 
194  if (selectedIndex == -1)
195  {
196  // Currently selected object is not in list, but nothing else was selected so no change.
197  return;
198  }
199  else if (selectedIndex == 0)
200  {
201  result = null; // Null option
202  }
203  else
204  {
205  result = objectList[selectedIndex - 1];
206  }
207 
208  property.objectReferenceValue = result;
209  }
