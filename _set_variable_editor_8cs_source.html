102  case SetOperator.Add:

104  break;
105  case SetOperator.Subtract:
106  operatorsList.Add(new GUIContent("-="));
107  break;
108  case SetOperator.Multiply:
109  operatorsList.Add(new GUIContent("*="));
110  break;
111  case SetOperator.Divide:
112  operatorsList.Add(new GUIContent("\\="));
113  break;
114  default:
115  Debug.LogError("The " + setOperator.ToString() + " operator has no matching GUIContent.");
116  break;
117  }
118  }
119 
120  // Get previously selected operator

122  if (selectedIndex < 0)
123  {

128
