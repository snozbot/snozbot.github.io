36  public static void VariableField(SerializedProperty property,
37  GUIContent label,
38  Flowchart flowchart,
39  string defaultText,
40  Func<Variable, bool> filter,
41  Func<string, int, string[], int> drawer = null)
42  {
43  List<string> variableKeys = new List<string>();
44  List<Variable> variableObjects = new List<Variable>();
45 
46  variableKeys.Add(defaultText);
47  variableObjects.Add(null);
48 
49  List<Variable> variables = flowchart.Variables;
50  int index = 0;
51  int selectedIndex = 0;
52 
53  Variable selectedVariable = property.objectReferenceValue as Variable;
54 
55  // When there are multiple Flowcharts in a scene with variables, switching
56  // between the Flowcharts can cause the wrong variable property
57  // to be inspected for a single frame. This has the effect of causing private
58  // variable references to be set to null when inspected. When this condition
59  // occurs we just skip displaying the property for this frame.
60  if (selectedVariable != null &&
61  selectedVariable.gameObject != flowchart.gameObject &&
62  selectedVariable.Scope == VariableScope.Private)
63  {
64  property.objectReferenceValue = null;
65  return;
66  }
67 
68  foreach (Variable v in variables)
69  {
70  if (filter != null)
71  {
72  if (!filter(v))
73  {
74  continue;
75  }
76  }
77 
78  variableKeys.Add(v.Key);
79  variableObjects.Add(v);
80 
81  index++;
82 
83  if (v == selectedVariable)
84  {
85  selectedIndex = index;
86  }
87  }
88 
89  Flowchart[] fsList = GameObject.FindObjectsOfType<Flowchart>();
90  foreach (Flowchart fs in fsList)
91  {
92  if (fs == flowchart)
93  {
94  continue;
95  }
96 
97  List<Variable> publicVars = fs.GetPublicVariables();
98  foreach (Variable v in publicVars)
99  {
100  if (filter != null)
101  {
102  if (!filter(v))
103  {
104  continue;
105  }
106  }
107 
108  variableKeys.Add(fs.name + " / " + v.Key);
109  variableObjects.Add(v);
110 
111  index++;
112 
113  if (v == selectedVariable)
114  {
115  selectedIndex = index;
116  }
117  }
118  }
119 
120  if (drawer == null)
121  {
122  selectedIndex = EditorGUILayout.Popup(label.text, selectedIndex, variableKeys.ToArray());
123  }
124  else
125  {
126  selectedIndex = drawer(label.text, selectedIndex, variableKeys.ToArray());
127  }
128 
129  property.objectReferenceValue = variableObjects[selectedIndex];
130  }
131  }

137  public override void OnGUI (Rect position, SerializedProperty property, GUIContent label)
138  {
139  VariablePropertyAttribute variableProperty = attribute as VariablePropertyAttribute;
140  if (variableProperty == null)
141  {
142  return;
143  }
144 
145  EditorGUI.BeginProperty(position, label, property);
146 
147  // Filter the variables by the types listed in the VariableProperty attribute
148  Func<Variable, bool> compare = v =>
149  {
150  if (v == null)
151  {
152  return false;
153  }
154 
155  if (variableProperty.VariableTypes.Length == 0)
156  {
157  return true;
158  }
159 
160  return variableProperty.VariableTypes.Contains<System.Type>(v.GetType());
161  };
162 
163  VariableEditor.VariableField(property,
164  label,
165  FlowchartWindow.GetFlowchart(),
166  variableProperty.defaultText,
167  compare,
168  (s,t,u) => (EditorGUI.Popup(position, s, t, u)));

174  public class VariableDataDrawer<T> : PropertyDrawer where T : Variable
175  {
176 
177  public override void OnGUI (Rect position, SerializedProperty property, GUIContent label)
178  {
179  EditorGUI.BeginProperty(position, label, property);
180 
181  // The variable reference and data properties must follow the naming convention 'typeRef', 'typeVal'
182 
183  VariableInfoAttribute typeInfo = VariableEditor.GetVariableInfo(typeof(T));
184  if (typeInfo == null)
185  {
186  return;
187  }
188 
189  string propNameBase = typeInfo.VariableType;
190  propNameBase = Char.ToLowerInvariant(propNameBase[0]) + propNameBase.Substring(1);
191 
192  SerializedProperty referenceProp = property.FindPropertyRelative(propNameBase + "Ref");
193  SerializedProperty valueProp = property.FindPropertyRelative(propNameBase + "Val");
194 
195  if (referenceProp == null || valueProp == null)
196  {
197  return;
198  }
199 
200  Command command = property.serializedObject.targetObject as Command;
201  if (command == null)
202  {
203  return;
204  }
205 
206  var flowchart = command.GetFlowchart() as Flowchart;
207  if (flowchart == null)
208  {
209  return;
210  }
211 
212  if (EditorGUI.GetPropertyHeight(valueProp, label) > EditorGUIUtility.singleLineHeight)
213  {
214  DrawMultiLineProperty(position, label, referenceProp, valueProp, flowchart);
215  }
216  else
217  {
218  DrawSingleLineProperty(position, label, referenceProp, valueProp, flowchart);
219  }
220 
221  EditorGUI.EndProperty();
222  }

266  public override float GetPropertyHeight(SerializedProperty property, GUIContent label)
267  {
268  VariableInfoAttribute typeInfo = VariableEditor.GetVariableInfo(typeof(T));
269  if (typeInfo == null)
270  {
271  return EditorGUIUtility.singleLineHeight;
272  }
273 
274  string propNameBase = typeInfo.VariableType;
275  propNameBase = Char.ToLowerInvariant(propNameBase[0]) + propNameBase.Substring(1);
276 
277  SerializedProperty referenceProp = property.FindPropertyRelative(propNameBase + "Ref");
278 
279  if (referenceProp.objectReferenceValue != null)
280  {
281  return EditorGUIUtility.singleLineHeight;
282  }
283 
284  SerializedProperty valueProp = property.FindPropertyRelative(propNameBase + "Val");
285  return EditorGUI.GetPropertyHeight(valueProp, label);
286  }
