497  protected virtual IEnumerator DoWords(List<string> paramList, TokenType previousTokenType)
498  {
499  if (!CheckParamCount(paramList, 1))
500  {
501  yield break;
502  }
503 
504  string param = paramList[0].Replace("\\n", "\n");
505 
506  // Trim whitespace after a {wc} or {c} tag
507  if (previousTokenType == TokenType.WaitForInputAndClear ||
508  previousTokenType == TokenType.Clear)
509  {
510  param = param.TrimStart(' ', '\t', '\r', '\n');
511  }
512 
513  // Start with the visible portion of any existing displayed text.
514  string startText = "";
515  if (visibleCharacterCount > 0 &&
516  visibleCharacterCount <= Text.Length)
517  {
518  startText = Text.Substring(0, visibleCharacterCount);
519  }
520 
521  UpdateOpenMarkup();
522  UpdateCloseMarkup();
523 
524  float timeAccumulator = Time.deltaTime;
525 
526  for (int i = 0; i < param.Length + 1; ++i)
527  {
528  // Exit immediately if the exit flag has been set
529  if (exitFlag)
530  {
531  break;
532  }
533 
534  // Pause mid sentence if Paused is set
535  while (Paused)
536  {
537  yield return null;
538  }
539 
540  PartitionString(writeWholeWords, param, i);
541  ConcatenateString(startText);
542  Text = outputString.ToString();
543 
544  NotifyGlyph();
545 
546  // No delay if user has clicked and Instant Complete is enabled
547  if (instantComplete && inputFlag)
548  {
549  continue;
550  }
551 
552  // Punctuation pause
553  if (leftString.Length > 0 &&
554  rightString.Length > 0 &&
555  IsPunctuation(leftString.ToString(leftString.Length - 1, 1)[0]))
556  {
557  yield return StartCoroutine(DoWait(currentPunctuationPause));
558  }
559 
560  // Delay between characters
561  if (currentWritingSpeed > 0f)
562  {
563  if (timeAccumulator > 0f)
564  {
565  timeAccumulator -= 1f / currentWritingSpeed;
566  }
567  else
568  {
569  yield return new WaitForSeconds(1f / currentWritingSpeed);
570  }
571  }
572  }
573  }

575  protected virtual void PartitionString(bool wholeWords, string inputString, int i)
576  {
577  leftString.Length = 0;
578  rightString.Length = 0;
579 
580  // Reached last character
581  leftString.Append(inputString);
582  if (i >= inputString.Length)
583  {
584  return;
585  }
586 
587  rightString.Append(inputString);
588 
589  if (wholeWords)
590  {
591  // Look ahead to find next whitespace or end of string
592  for (int j = i; j < inputString.Length + 1; ++j)
593  {
594  if (j == inputString.Length || Char.IsWhiteSpace(inputString[j]))
595  {
596  leftString.Length = j;
597  rightString.Remove(0, j);
598  break;
599  }
600  }
601  }
602  else
603  {
604  leftString.Remove(i, inputString.Length - i);
605  rightString.Remove(0, i);
606  }
607  }
