499  protected virtual IEnumerator DoWords(List<string> paramList, TokenType previousTokenType)
500  {
501  if (!CheckParamCount(paramList, 1))
502  {
503  yield break;
504  }
505 
506  string param = paramList[0];
507 
508  // Trim whitespace after a {wc} or {c} tag
509  if (previousTokenType == TokenType.WaitForInputAndClear ||
510  previousTokenType == TokenType.Clear)
511  {
512  param = param.TrimStart(' ', '\t', '\r', '\n');
513  }
514 
515  // Start with the visible portion of any existing displayed text.
516  string startText = "";
517  if (visibleCharacterCount > 0 &&
518  visibleCharacterCount <= Text.Length)
519  {
520  startText = Text.Substring(0, visibleCharacterCount);
521  }
522 
523  UpdateOpenMarkup();
524  UpdateCloseMarkup();
525 
526  float timeAccumulator = Time.deltaTime;
527 
528  for (int i = 0; i < param.Length + 1; ++i)
529  {
530  // Exit immediately if the exit flag has been set
531  if (exitFlag)
532  {
533  break;
534  }
535 
536  // Pause mid sentence if Paused is set
537  while (Paused)
538  {
539  yield return null;
540  }
541 
542  PartitionString(writeWholeWords, param, i);
543  ConcatenateString(startText);
544  Text = outputString.ToString();
545 
546  NotifyGlyph();
547 
548  // No delay if user has clicked and Instant Complete is enabled
549  if (instantComplete && inputFlag)
550  {
551  continue;
552  }
553 
554  // Punctuation pause
555  if (leftString.Length > 0 &&
556  rightString.Length > 0 &&
557  IsPunctuation(leftString.ToString(leftString.Length - 1, 1)[0]))
558  {
559  yield return StartCoroutine(DoWait(currentPunctuationPause));
560  }
561 
562  // Delay between characters
563  if (currentWritingSpeed > 0f)
564  {
565  if (timeAccumulator > 0f)
566  {
567  timeAccumulator -= 1f / currentWritingSpeed;
568  }
569  else
570  {
571  yield return new WaitForSeconds(1f / currentWritingSpeed);
572  }
573  }
574  }
575  }

577  protected virtual void PartitionString(bool wholeWords, string inputString, int i)
578  {
579  leftString.Length = 0;
580  rightString.Length = 0;
581 
582  // Reached last character
583  leftString.Append(inputString);
584  if (i >= inputString.Length)
585  {
586  return;
587  }
588 
589  rightString.Append(inputString);
590 
591  if (wholeWords)
592  {
593  // Look ahead to find next whitespace or end of string
594  for (int j = i; j < inputString.Length + 1; ++j)
595  {
596  if (j == inputString.Length || Char.IsWhiteSpace(inputString[j]))
597  {
598  leftString.Length = j;
599  rightString.Remove(0, j);
600  break;
601  }
602  }
603  }
604  else
605  {
606  leftString.Remove(i, inputString.Length - i);
607  rightString.Remove(0, i);
608  }
609  }
