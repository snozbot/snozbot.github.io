205  private void ResizeScrollView(Flowchart flowchart)
206  {
207  Rect cursorChangeRect = new Rect(0, flowchart.BlockViewHeight + 1, Screen.width, 4f);
208 
209  EditorGUIUtility.AddCursorRect(cursorChangeRect, MouseCursor.ResizeVertical);
210 
211  if (Event.current.type == EventType.mouseDown && cursorChangeRect.Contains(Event.current.mousePosition))
212  {
213  resize = true;
214  }
215 
216  if (resize)
217  {
218  Undo.RecordObject(flowchart, "Resize view");
219  flowchart.BlockViewHeight = Event.current.mousePosition.y;
220  }
221 
222  ClampBlockViewHeight(flowchart);
223 
224  // Stop resizing if mouse is outside inspector window.
225  // This isn't standard Unity UI behavior but it is robust and safe.
226  if (resize && Event.current.type == EventType.mouseDrag)
227  {
228  Rect windowRect = new Rect(0, 0, Screen.width, Screen.height);
229  if (!windowRect.Contains(Event.current.mousePosition))
230  {
231  resize = false;
232  }
233  }
234 
235  if (Event.current.type == EventType.MouseUp)
236  {
237  resize = false;
238  }
239  }

241  protected virtual void ClampBlockViewHeight(Flowchart flowchart)
242  {
243  // Screen.height seems to temporarily reset to 480 for a single frame whenever a command like
244  // Copy, Paste, etc. happens. Only clamp the block view height when one of these operations is not occuring.
245 
246  if (Event.current.commandName != "")
247  {
248  clamp = false;
249  }
250 
251  if (clamp)
252  {
253  // Make sure block view is always clamped to visible area
254  float height = flowchart.BlockViewHeight;
255  height = Mathf.Max(200, height);
256  height = Mathf.Min(Screen.height - 200,height);
257  flowchart.BlockViewHeight = height;
258  }
259 
260  if (Event.current.type == EventType.Repaint)
261  {
262  clamp = true;
263  }
264  }
