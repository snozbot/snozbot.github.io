76  private MethodInfo ShowMethods(SerializedObject objTarget, GameObject gameObject, string component)
77  {
78  MethodInfo result = null;
79 
80  var targetMethodProp = objTarget.FindProperty("targetMethod");
81  var targetMethodTextProp = objTarget.FindProperty("targetMethodText");
82  var methodParametersProp = objTarget.FindProperty("methodParameters");
83  var showInheritedProp = objTarget.FindProperty("showInherited");
84  var saveReturnValueProp = objTarget.FindProperty("saveReturnValue");
85  var returnValueKeyProp = objTarget.FindProperty("returnValueVariableKey");
86 
87  var objComponent = gameObject.GetComponent(ReflectionHelper.GetType(component));
88  var bindingFlags = BindingFlags.Default | BindingFlags.Public | BindingFlags.Instance;
89 
90  if (!showInheritedProp.boolValue)
91  {
92  bindingFlags |= BindingFlags.DeclaredOnly;
93  }
94 
95  if (objComponent != null)
96  {
97  var objMethods = objComponent.GetType().GetMethods(bindingFlags);
98  var methods = (from objMethod in objMethods where !objMethod.IsSpecialName select objMethod).ToList(); // filter out the getter/setter methods
99  var methodText = (from objMethod in methods select objMethod.Name + FormatParameters(objMethod.GetParameters()) + ": " + objMethod.ReturnType.Name).ToList();
100  int index = methodText.IndexOf(targetMethodTextProp.stringValue);
101 
102  index = EditorGUILayout.Popup("Target Method", index, methodText.ToArray());
103 
104  EditorGUILayout.PropertyField(showInheritedProp);
105 
106  if (index != -1)
107  {
108  if (targetMethodTextProp.stringValue != methodText[index])
109  {
110  // reset
111  methodParametersProp.ClearArray();
112  methodParametersProp.arraySize = methods[index].GetParameters().Length;
113 
114  saveReturnValueProp.boolValue = false;
115  returnValueKeyProp.stringValue = null;
116  }
117 
118  targetMethodTextProp.stringValue = methodText[index];
119  targetMethodProp.stringValue = methods[index].Name;
120 
121  result = methods[index];
122  }
123  else
124  {
125  targetMethodTextProp.stringValue = null;
126  targetMethodProp.stringValue = null;
127  }
128 
129  objTarget.ApplyModifiedProperties();
130  }
131 
132  return result;
133  }
