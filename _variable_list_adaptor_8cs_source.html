146  public void DrawItem(Rect position, int index, bool selected, bool focused)
147  {
148  Variable variable = GetVarAt(index);// this[index].objectReferenceValue as Variable;
149 
150  if (variable == null)
151  {
152  return;
153  }
154 
155  for (int i = 0; i < 4; ++i)
156  {
157  itemRects[i] = position;
158  itemRects[i].width = itemWidths[i] - 5;
159 
160  for (int j = 0; j < i; ++j)
161  {
162  itemRects[i].x += itemWidths[j];
163  }
164  }
165 
166  VariableInfoAttribute variableInfo = VariableEditor.GetVariableInfo(variable.GetType());
167  if (variableInfo == null)
168  {
169  return;
170  }
171 
172  var flowchart = TargetFlowchart;
173  if (flowchart == null)
174  {
175  return;
176  }
177 
178  // Highlight if an active or selected command is referencing this variable
179  bool highlight = false;
180  if (flowchart.SelectedBlock != null)
181  {
182  if (Application.isPlaying && flowchart.SelectedBlock.IsExecuting())
183  {
184  highlight = flowchart.SelectedBlock.ActiveCommand.IsVariableReferenced(variable);
185  }
186  else if (!Application.isPlaying && flowchart.SelectedCommands.Count > 0)
187  {
188  foreach (Command selectedCommand in flowchart.SelectedCommands)
189  {
190  if (selectedCommand == null)
191  {
192  continue;
193  }
194 
195  if (selectedCommand.IsVariableReferenced(variable))
196  {
197  highlight = true;
198  break;
199  }
200  }
201  }
202  }
203 
204  if (highlight)
205  {
206  GUI.backgroundColor = Color.green;
207  GUI.Box(position, "");
208  }
209 
210  string key = variable.Key;
211  VariableScope scope = variable.Scope;
212 
213  // To access properties in a monobehavior, you have to new a SerializedObject
214  // http://answers.unity3d.com/questions/629803/findrelativeproperty-never-worked-for-me-how-does.html
215  SerializedObject variableObject = new SerializedObject(variable);
216 
217  variableObject.Update();
218 
219  GUI.Label(itemRects[0], variableInfo.VariableType);
220 
221  SerializedProperty keyProp = variableObject.FindProperty("key");
222  SerializedProperty defaultProp = variableObject.FindProperty("value");
223  SerializedProperty scopeProp = variableObject.FindProperty("scope");
224 
225 
226  EditorGUI.BeginChangeCheck();
227  key = EditorGUI.TextField(itemRects[1], variable.Key);
228  if (EditorGUI.EndChangeCheck())
229  {
230  keyProp.stringValue = flowchart.GetUniqueVariableKey(key, variable);
231  }
232 
233  bool isGlobal = scopeProp.enumValueIndex == (int)VariableScope.Global;
234 
235 
236  if (isGlobal && Application.isPlaying)
237  {
238  var res = FungusManager.Instance.GlobalVariables.GetVariable(keyProp.stringValue);
239  if (res != null)
240  {
241  SerializedObject globalValue = new SerializedObject(res);
242  var globalValProp = globalValue.FindProperty("value");
243 
244  var prevEnabled = GUI.enabled;
245  GUI.enabled = false;
246 
247  EditorGUI.PropertyField(itemRects[2], globalValProp, emptyGUIContent);
248 
249  GUI.enabled = prevEnabled;
250  }
251  }
252  else
253  {
254  EditorGUI.PropertyField(itemRects[2], defaultProp, emptyGUIContent);
255  }
256 
257

259  scopeProp.enumValueIndex = (int)scope;
260

263  GUI.backgroundColor = Color.white;
264  }
