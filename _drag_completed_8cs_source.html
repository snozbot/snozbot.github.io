124  public virtual void OnDragCompleted(Draggable2D draggableObject)
125  {

127  overTarget)
128  {
129  // Assume that the player will have to do perform another drag and drop operation
130  // to complete the drag again. This is necessary because we don't get an OnDragExited if the
