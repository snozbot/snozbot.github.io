127  protected virtual void DoEndDrag()
128  {
129  if (!dragEnabled)
130  {
131  return;
132  }
133 
134  var eventDispatcher = FungusManager.Instance.EventDispatcher;
135  bool dragCompleted = false;
136 
137  for (int i = 0; i < dragCompletedHandlers.Count; i++)
138  {
139  var handler = dragCompletedHandlers[i];
140  if (handler != null && handler.DraggableObject == this)
141  {
142  if (handler.IsOverTarget())
143  {
144  dragCompleted = true;
145 
146  eventDispatcher.Raise(new DragCompleted.DragCompletedEvent(this));
147  }
148  }
149  }
150 
151  if (!dragCompleted)
152  {
153  eventDispatcher.Raise(new DragCancelled.DragCancelledEvent(this));
154 
155  if (returnOnCancelled)
156  {
157  LeanTween.move(gameObject, startingPosition, returnDuration).setEase(LeanTweenType.easeOutExpo);
158  }
159  }
160  else if(returnOnCompleted)
161  {
162  LeanTween.move(gameObject, startingPosition, returnDuration).setEase(LeanTweenType.easeOutExpo);
163  }
164  }
