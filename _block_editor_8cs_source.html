691  protected static List<KeyValuePair<System.Type,CommandInfoAttribute>> GetFilteredCommandInfoAttribute(List<System.Type> menuTypes)
692  {
693  Dictionary<string, KeyValuePair<System.Type, CommandInfoAttribute>> filteredAttributes = new Dictionary<string, KeyValuePair<System.Type, CommandInfoAttribute>>();
694 
695  foreach (System.Type type in menuTypes)
696  {
697  object[] attributes = type.GetCustomAttributes(false);
698  foreach (object obj in attributes)
699  {
700  CommandInfoAttribute infoAttr = obj as CommandInfoAttribute;
701  if (infoAttr != null)
702  {
703  string dictionaryName = string.Format("{0}/{1}", infoAttr.Category, infoAttr.CommandName);
704 
705  int existingItemPriority = -1;
706  if (filteredAttributes.ContainsKey(dictionaryName))
707  {
708  existingItemPriority = filteredAttributes[dictionaryName].Value.Priority;
709  }
710 
711  if (infoAttr.Priority > existingItemPriority)
712  {
713  KeyValuePair<System.Type, CommandInfoAttribute> keyValuePair = new KeyValuePair<System.Type, CommandInfoAttribute>(type, infoAttr);
714  filteredAttributes[dictionaryName] = keyValuePair;
715  }
716  }
717  }
718  }
719  return filteredAttributes.Values.ToList<KeyValuePair<System.Type,CommandInfoAttribute>>();
720  }
