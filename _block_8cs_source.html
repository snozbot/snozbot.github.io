199  public virtual IEnumerator Execute(int commandIndex = 0, Action onComplete = null)
200  {
201  if (executionState != ExecutionState.Idle)
202  {
203  yield break;
204  }
205 
206  if (!executionInfoSet)
207  {
208  SetExecutionInfo();
209  }
210 
211  executionCount++;
212 
213  var flowchart = GetFlowchart();
214  executionState = ExecutionState.Executing;
215  BlockSignals.DoBlockStart(this);
216 
217  #if UNITY_EDITOR
218  // Select the executing block & the first command
219  flowchart.SelectedBlock = this;
220  if (commandList.Count > 0)
221  {
222  flowchart.ClearSelectedCommands();
223  flowchart.AddSelectedCommand(commandList[0]);
224  }
225  #endif
226 
227  jumpToCommandIndex = commandIndex;
228 
229  int i = 0;
230  while (true)
231  {
232  // Executing commands specify the next command to skip to by setting jumpToCommandIndex using Command.Continue()
233  if (jumpToCommandIndex > -1)
234  {
235  i = jumpToCommandIndex;
236  jumpToCommandIndex = -1;
237  }
238 
239  // Skip disabled commands, comments and labels
240  while (i < commandList.Count &&
241  (!commandList[i].enabled ||
242  commandList[i].GetType() == typeof(Comment) ||
243  commandList[i].GetType() == typeof(Label)))
244  {
245  i = commandList[i].CommandIndex + 1;
246  }
247 
248  if (i >= commandList.Count)
249  {
250  break;
251  }
252 
253  // The previous active command is needed for if / else / else if commands
254  if (activeCommand == null)
255  {
256  previousActiveCommandIndex = -1;
257  }
258  else
259  {
260  previousActiveCommandIndex = activeCommand.CommandIndex;
261  }
262 
263  var command = commandList[i];
264  activeCommand = command;
265 
266  if (flowchart.IsActive())
267  {
268  // Auto select a command in some situations
269  if ((flowchart.SelectedCommands.Count == 0 && i == 0) ||
270  (flowchart.SelectedCommands.Count == 1 && flowchart.SelectedCommands[0].CommandIndex == previousActiveCommandIndex))
271  {
272  flowchart.ClearSelectedCommands();
273  flowchart.AddSelectedCommand(commandList[i]);
274  }
275  }
276 
277  command.IsExecuting = true;
278  // This icon timer is managed by the FlowchartWindow class, but we also need to
279  // set it here in case a command starts and finishes execution before the next window update.
280  command.ExecutingIconTimer = Time.realtimeSinceStartup + FungusConstants.ExecutingIconFadeTime;
281  BlockSignals.DoCommandExecute(this, command, i, commandList.Count);
282  command.Execute();
283 
284  // Wait until the executing command sets another command to jump to via Command.Continue()
285  while (jumpToCommandIndex == -1)
286  {
287  yield return null;
288  }
289 
290  #if UNITY_EDITOR
291  if (flowchart.StepPause > 0f)
292  {
293  yield return new WaitForSeconds(flowchart.StepPause);
294  }
295  #endif
296 
297  command.IsExecuting = false;
298  }
299 
300  executionState = ExecutionState.Idle;
301  activeCommand = null;
302  BlockSignals.DoBlockEnd(this);
303 
304  if (onComplete != null)
305  {
306  onComplete();
307  }
308  }
