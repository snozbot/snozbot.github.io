186  protected virtual void AddCSVDataItems(Dictionary<string, TextItem> textItems, string csvData)
187  {
188  CsvParser csvParser = new CsvParser();
189  string[][] csvTable = csvParser.Parse(csvData);
190 
191  if (csvTable.Length <= 1)
192  {
193  // No data rows in file
194  return;
195  }
196 
197  // Parse header row
198  string[] columnNames = csvTable[0];
199 
200  for (int i = 1; i < csvTable.Length; ++i)
201  {
202  string[] fields = csvTable[i];
203  if (fields.Length < 3)
204  {
205  // No standard text or localized string fields present
206  continue;
207  }
208 
209  string stringId = fields[0];
210 
211  if (!textItems.ContainsKey(stringId))
212  {
213  if (stringId.StartsWith("CHARACTER.") ||
214  stringId.StartsWith("SAY.") ||
215  stringId.StartsWith("MENU.") ||
216  stringId.StartsWith("WRITE.") ||
217  stringId.StartsWith("SETTEXT."))
218  {
219  // If it's a 'built-in' type this probably means that item has been deleted from its flowchart,
220  // so there's no need to add a text item for it.
221  continue;
222  }
223 
224  // Key not found. Assume it's a custom string that we want to retain, so add a text item for it.
225  TextItem newTextItem = new TextItem();
226  newTextItem.description = CSVSupport.Unescape(fields[1]);
227  newTextItem.standardText = CSVSupport.Unescape(fields[2]);
228  textItems[stringId] = newTextItem;
229  }
230 
231  TextItem textItem = textItems[stringId];
232 
233  for (int j = 3; j < fields.Length; ++j)
234  {
235  if (j >= columnNames.Length)
236  {
237  continue;
238  }
239  string languageCode = columnNames[j];
240  string languageEntry = CSVSupport.Unescape(fields[j]);
241 
242  if (languageEntry.Length > 0)
243  {
244  textItem.localizedStrings[languageCode] = languageEntry;
245  }
246  }
247  }

298  public virtual string GetCSVData()
299  {
300  // Collect all the text items present in the scene
301  Dictionary<string, TextItem> textItems = FindTextItems();
302 
303  // Update text items with localization data from CSV file
304  if (localizationFile != null &&
305  localizationFile.text.Length > 0)
306  {
307  AddCSVDataItems(textItems, localizationFile.text);
308  }
309 
310  // Build CSV header row and a list of the language codes currently in use
311  string csvHeader = "Key,Description,Standard";
312  var languageCodes = new List<string>();
313  var values = textItems.Values;
314  foreach (var textItem in values)
315  {
316  foreach (string languageCode in textItem.localizedStrings.Keys)
317  {
318  if (!languageCodes.Contains(languageCode))
319  {
320  languageCodes.Add(languageCode);
321  csvHeader += "," + languageCode;
322  }
323  }
324  }
325 
326  // Build the CSV file using collected text items
327  int rowCount = 0;
328  string csvData = csvHeader + "\n";
329  var keys = textItems.Keys;
330  foreach (var stringId in keys)
331  {
332  TextItem textItem = textItems[stringId];
333 
334  string row = CSVSupport.Escape(stringId);
335  row += "," + CSVSupport.Escape(textItem.description);
336  row += "," + CSVSupport.Escape(textItem.standardText);
337 
338  for (int i = 0; i < languageCodes.Count; i++)
339  {
340  var languageCode = languageCodes[i];
341  if (textItem.localizedStrings.ContainsKey(languageCode))
342  {
343  row += "," + CSVSupport.Escape(textItem.localizedStrings[languageCode]);
344  }
345  else
346  {
347  row += ",";
348  // Empty field
349  }
350  }
351 
352  csvData += row + "\n";
353  rowCount++;
354  }
355 
356  notificationText = "Exported " + rowCount + " localization text items.";
357 
358  return csvData;
359  }

365  public virtual void SetActiveLanguage(string languageCode, bool forceUpdateSceneText = false)
366  {
367  if (!Application.isPlaying)
368  {
369  // This function should only ever be called when the game is playing (not in editor).
370  return;
371  }
372 
373  if (localizationFile == null)
374  {
375  // No localization file set
376  return;
377  }
378 
379  localizedStrings.Clear();
380 
381  CsvParser csvParser = new CsvParser();
382  string[][] csvTable = csvParser.Parse(localizationFile.text);
383 
384  if (csvTable.Length <= 1)
385  {
386  // No data rows in file
387  return;
388  }
389 
390  // Parse header row
391  string[] columnNames = csvTable[0];
392 
393  if (columnNames.Length < 3)
394  {
395  // No languages defined in CSV file
396  return;
397  }
398 
399  // First assume standard text column and then look for a matching language column
400  int languageIndex = 2;
401  for (int i = 3; i < columnNames.Length; ++i)
402  {
403  if (columnNames[i] == languageCode)
404  {
405  languageIndex = i;
406  break;
407  }
408  }
409 
410  if (languageIndex == 2)
411  {
412  // Using standard text column
413  // Add all strings to the localized strings dict, but don't replace standard text in the scene.
414  // This allows string substitution to work for both standard and localized text strings.
415  for (int i = 1; i < csvTable.Length; ++i)
416  {
417  string[] fields = csvTable[i];
418  if (fields.Length < 3)
419  {
420  continue;
421  }
422 
423  localizedStrings[fields[0]] = fields[languageIndex];
424  }
425 
426  // Early out unless we've been told to force the scene text to update.
427  // This happens when the Set Language command is used to reset back to the standard language.
428  if (!forceUpdateSceneText)
429  {
430  return;
431  }
432  }
433 
434  // Using a localized language text column
435  // 1. Add all localized text to the localized strings dict
436  // 2. Update all scene text properties with localized versions
437  for (int i = 1; i < csvTable.Length; ++i)
438  {
439  string[] fields = csvTable[i];
440 
441  if (fields.Length < languageIndex + 1)
442  {
443  continue;
444  }
445 
446  string stringId = fields[0];
447  string languageEntry = CSVSupport.Unescape(fields[languageIndex]);
448 
449  if (languageEntry.Length > 0)
450  {
451  localizedStrings[stringId] = languageEntry;
452  PopulateTextProperty(stringId, languageEntry);
453  }
454  }
455  }

508  public virtual void SetStandardText(string textData)
509  {
510  var lines = textData.Split('\n');
511 
512  int updatedCount = 0;
513 
514  string stringId = "";
515  string buffer = "";
516  for (int i = 0; i < lines.Length; i++)
517  {
518  // Check for string id line
519  var line = lines[i];
520  if (line.StartsWith("#"))
521  {
522  if (stringId.Length > 0)
523  {
524  // Write buffered text to the appropriate text property
525  if (PopulateTextProperty(stringId, buffer.Trim()))
526  {
527  updatedCount++;
528  }
529  }
530  // Set the string id for the follow text lines
531  stringId = line.Substring(1, line.Length - 1);
532  buffer = "";
533  }
534  else
535  {
536  buffer += line + "\n";
537  }
538  }
539 
540  // Handle last buffered entry
541  if (stringId.Length > 0)
542  {
543  if (PopulateTextProperty(stringId, buffer.Trim()))
544  {
545  updatedCount++;
546  }
547  }
548 
549  notificationText = "Updated " + updatedCount + " standard text items.";
550  }

556  public virtual bool SubstituteStrings(StringBuilder input)
557  {
558  // This method could be called from the Start method of another component, so we
559  // may need to initilize the localization system.
560  Init();
561 
562  // Instantiate the regular expression object.
563  Regex r = new Regex("{\\$.*?}");
564 
565  bool modified = false;
566 
567  // Match the regular expression pattern against a text string.
568  var results = r.Matches(input.ToString());
569  for (int i = 0; i < results.Count; i++)
570  {
571  Match match = results[i];
572  string key = match.Value.Substring(2, match.Value.Length - 3);
573  // Next look for matching localized string
574  string localizedString = Localization.GetLocalizedString(key);
575  if (localizedString != null)
576  {
577  input.Replace(match.Value, localizedString);
578  modified = true;
579  }
580  }
581 
582  return modified;
583  }
