171  protected virtual void AddCSVDataItems(Dictionary<string, TextItem> textItems, string csvData)
172  {
173  CsvParser csvParser = new CsvParser();
174  string[][] csvTable = csvParser.Parse(csvData);
175 
176  if (csvTable.Length <= 1)
177  {
178  // No data rows in file
179  return;
180  }
181 
182  // Parse header row
183  string[] columnNames = csvTable[0];
184 
185  for (int i = 1; i < csvTable.Length; ++i)
186  {
187  string[] fields = csvTable[i];
188  if (fields.Length < 3)
189  {
190  // No standard text or localized string fields present
191  continue;
192  }
193 
194  string stringId = fields[0];
195 
196  if (!textItems.ContainsKey(stringId))
197  {
198  if (stringId.StartsWith("CHARACTER.") ||
199  stringId.StartsWith("SAY.") ||
200  stringId.StartsWith("MENU.") ||
201  stringId.StartsWith("WRITE.") ||
202  stringId.StartsWith("SETTEXT."))
203  {
204  // If it's a 'built-in' type this probably means that item has been deleted from its flowchart,
205  // so there's no need to add a text item for it.
206  continue;
207  }
208 
209  // Key not found. Assume it's a custom string that we want to retain, so add a text item for it.
210  TextItem newTextItem = new TextItem();
211  newTextItem.description = CSVSupport.Unescape(fields[1]);
212  newTextItem.standardText = CSVSupport.Unescape(fields[2]);
213  textItems[stringId] = newTextItem;
214  }
215 
216  TextItem textItem = textItems[stringId];
217 
218  for (int j = 3; j < fields.Length; ++j)
219  {
220  if (j >= columnNames.Length)
221  {
222  continue;
223  }
224  string languageCode = columnNames[j];
225  string languageEntry = CSVSupport.Unescape(fields[j]);
226 
227  if (languageEntry.Length > 0)
228  {
229  textItem.localizedStrings[languageCode] = languageEntry;
230  }
231  }
232  }

283  public virtual string GetCSVData()
284  {
285  // Collect all the text items present in the scene
286  Dictionary<string, TextItem> textItems = FindTextItems();
287 
288  // Update text items with localization data from CSV file
289  if (localizationFile != null &&
290  localizationFile.text.Length > 0)
291  {
292  AddCSVDataItems(textItems, localizationFile.text);
293  }
294 
295  // Build CSV header row and a list of the language codes currently in use
296  string csvHeader = "Key,Description,Standard";
297  var languageCodes = new List<string>();
298  var values = textItems.Values;
299  foreach (var textItem in values)
300  {
301  foreach (string languageCode in textItem.localizedStrings.Keys)
302  {
303  if (!languageCodes.Contains(languageCode))
304  {
305  languageCodes.Add(languageCode);
306  csvHeader += "," + languageCode;
307  }
308  }
309  }
310 
311  // Build the CSV file using collected text items
312  int rowCount = 0;
313  string csvData = csvHeader + "\n";
314  var keys = textItems.Keys;
315  foreach (var stringId in keys)
316  {
317  TextItem textItem = textItems[stringId];
318 
319  string row = CSVSupport.Escape(stringId);
320  row += "," + CSVSupport.Escape(textItem.description);
321  row += "," + CSVSupport.Escape(textItem.standardText);
322 
323  for (int i = 0; i < languageCodes.Count; i++)
324  {
325  var languageCode = languageCodes[i];
326  if (textItem.localizedStrings.ContainsKey(languageCode))
327  {
328  row += "," + CSVSupport.Escape(textItem.localizedStrings[languageCode]);
329  }
330  else
331  {
332  row += ",";
333  // Empty field
334  }
335  }
336 
337  csvData += row + "\n";
338  rowCount++;
339  }
340 
341  notificationText = "Exported " + rowCount + " localization text items.";
342 
343  return csvData;
344  }

350  public virtual void SetActiveLanguage(string languageCode, bool forceUpdateSceneText = false)
351  {
352  if (!Application.isPlaying)
353  {
354  // This function should only ever be called when the game is playing (not in editor).
355  return;
356  }
357 
358  if (localizationFile == null)
359  {
360  // No localization file set
361  return;
362  }
363 
364  localizedStrings.Clear();
365 
366  CsvParser csvParser = new CsvParser();
367  string[][] csvTable = csvParser.Parse(localizationFile.text);
368 
369  if (csvTable.Length <= 1)
370  {
371  // No data rows in file
372  return;
373  }
374 
375  // Parse header row
376  string[] columnNames = csvTable[0];
377 
378  if (columnNames.Length < 3)
379  {
380  // No languages defined in CSV file
381  return;
382  }
383 
384  // First assume standard text column and then look for a matching language column
385  int languageIndex = 2;
386  for (int i = 3; i < columnNames.Length; ++i)
387  {
388  if (columnNames[i] == languageCode)
389  {
390  languageIndex = i;
391  break;
392  }
393  }
394 
395  if (languageIndex == 2)
396  {
397  // Using standard text column
398  // Add all strings to the localized strings dict, but don't replace standard text in the scene.
399  // This allows string substitution to work for both standard and localized text strings.
400  for (int i = 1; i < csvTable.Length; ++i)
401  {
402  string[] fields = csvTable[i];
403  if (fields.Length < 3)
404  {
405  continue;
406  }
407 
408  localizedStrings[fields[0]] = fields[languageIndex];
409  }
410 
411  // Early out unless we've been told to force the scene text to update.
412  // This happens when the Set Language command is used to reset back to the standard language.
413  if (!forceUpdateSceneText)
414  {
415  return;
416  }
417  }
418 
419  // Using a localized language text column
420  // 1. Add all localized text to the localized strings dict
421  // 2. Update all scene text properties with localized versions
422  for (int i = 1; i < csvTable.Length; ++i)
423  {
424  string[] fields = csvTable[i];
425 
426  if (fields.Length < languageIndex + 1)
427  {
428  continue;
429  }
430 
431  string stringId = fields[0];
432  string languageEntry = CSVSupport.Unescape(fields[languageIndex]);
433 
434  if (languageEntry.Length > 0)
435  {
436  localizedStrings[stringId] = languageEntry;
437  PopulateTextProperty(stringId, languageEntry);
438  }
439  }
440  }

493  public virtual void SetStandardText(string textData)
494  {
495  var lines = textData.Split('\n');
496 
497  int updatedCount = 0;
498 
499  string stringId = "";
500  string buffer = "";
501  for (int i = 0; i < lines.Length; i++)
502  {
503  // Check for string id line
504  var line = lines[i];
505  if (line.StartsWith("#"))
506  {
507  if (stringId.Length > 0)
508  {
509  // Write buffered text to the appropriate text property
510  if (PopulateTextProperty(stringId, buffer.Trim()))
511  {
512  updatedCount++;
513  }
514  }
515  // Set the string id for the follow text lines
516  stringId = line.Substring(1, line.Length - 1);
517  buffer = "";
518  }
519  else
520  {
521  buffer += line + "\n";
522  }
523  }
524 
525  // Handle last buffered entry
526  if (stringId.Length > 0)
527  {
528  if (PopulateTextProperty(stringId, buffer.Trim()))
529  {
530  updatedCount++;
531  }
532  }
533 
534  notificationText = "Updated " + updatedCount + " standard text items.";
535  }

541  public virtual bool SubstituteStrings(StringBuilder input)
542  {
543  // This method could be called from the Start method of another component, so we
544  // may need to initilize the localization system.
545  Init();
546 
547  // Instantiate the regular expression object.
548  Regex r = new Regex("{\\$.*?}");
549 
550  bool modified = false;
551 
552  // Match the regular expression pattern against a text string.
553  var results = r.Matches(input.ToString());
554  for (int i = 0; i < results.Count; i++)
555  {
556  Match match = results[i];
557  string key = match.Value.Substring(2, match.Value.Length - 3);
558  // Next look for matching localized string
559  string localizedString = Localization.GetLocalizedString(key);
560  if (localizedString != null)
561  {
562  input.Replace(match.Value, localizedString);
563  modified = true;
564  }
565  }
566 
567  return modified;
568  }
